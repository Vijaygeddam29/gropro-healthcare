app/models/models.py
from sqlalchemy import Column, Integer, String, Float, DateTime, ForeignKey
from sqlalchemy.ext.declarative import declarative_base
from datetime import datetime

Base = declarative_base()

class User(Base):
    __tablename__ = "users"
    id = Column(Integer, primary_key=True)
    name = Column(String)
    email = Column(String, unique=True)
    role = Column(String)  # doctor, accountant, advisor
    country = Column(String)
    county = Column(String)
    profession = Column(String)
    created_at = Column(DateTime, default=datetime.utcnow)

class FIC(Base):
    __tablename__ = "fics"
    id = Column(Integer, primary_key=True)
    user_id = Column(Integer, ForeignKey("users.id"))
    fic_name = Column(String)
    capital = Column(Float)
    corporation_tax = Column(Float)
    dividend_strategy = Column(String)
    created_at = Column(DateTime, default=datetime.utcnow)

class Investment(Base):
    __tablename__ = "investments"
    id = Column(Integer, primary_key=True)
    fic_id = Column(Integer, ForeignKey("fics.id"))
    type = Column(String)  # ETF, Property, Venture
    amount = Column(Float)
    start_date = Column(DateTime)
    projected_return = Column(Float)
    created_at = Column(DateTime, default=datetime.utcnow)

class Advisor(Base):
    __tablename__ = "advisors"
    id = Column(Integer, primary_key=True)
    name = Column(String)
    expertise = Column(String)
    verified = Column(String)
    contact_info = Column(String)

class Region(Base):
    __tablename__ = "regions"
    id = Column(Integer, primary_key=True)
    country = Column(String)
    county = Column(String)
    region_code = Column(String)
app/ai_agents/financial_mri.py
class FinancialMRI:
    """
    Calculates the MRI score (debt, savings, pension, investment efficiency)
    """
    def __init__(self, income, expenses, savings, pension, investments, debt):
        self.income = income
        self.expenses = expenses
        self.savings = savings
        self.pension = pension
        self.investments = investments
        self.debt = debt

    def score(self):
        # Simple scoring algorithm
        net_worth = self.savings + self.pension + self.investments - self.debt
        efficiency = net_worth / (self.income + 1e-6)
        # Scale to 0-100
        score = min(max(int(efficiency*100),0),100)
        return score

    def analysis(self):
        score = self.score()
        advice = []
        if self.debt > self.income * 0.5:
            advice.append("High debt, consider repayment strategies")
        if self.savings < self.income * 0.3:
            advice.append("Low cash buffer, consider increasing savings")
        if self.pension < self.income * 0.2:
            advice.append("Pension contributions low, optimize for tapering and LTA")
        return {"score": score, "advice": advice}

class ScenarioAI:
    """
    Generates A/B/C investment and tax scenarios
    """
    def __init__(self, fic, pension, tax_rate):
        self.fic = fic
        self.pension = pension
        self.tax_rate = tax_rate

    def generate_scenarios(self):
        scenarios = {
            "A": {"risk": "conservative", "projected_return": self.fic.capital*0.04, "tax_impact": self.fic.capital*self.tax_rate},
            "B": {"risk": "balanced", "projected_return": self.fic.capital*0.07, "tax_impact": self.fic.capital*self.tax_rate*0.9},
            "C": {"risk": "aggressive", "projected_return": self.fic.capital*0.12, "tax_impact": self.fic.capital*self.tax_rate*0.85}
        }
        return scenarios
app/ai_agents/pension_ai.py
class PensionAI:
    """
    Models lifetime allowance, tapering, and SIPP contributions
    """
    def __init__(self, pension_value, contribution, income, age):
        self.pension_value = pension_value
        self.contribution = contribution
        self.income = income
        self.age = age

    def model(self):
        # Basic tapering model
        lta_limit = 1_073_100  # UK 2026 LTA
        future_value = self.pension_value + self.contribution* (65 - self.age)
        taper_warning = future_value > lta_limit
        return {"future_value": future_value, "taper_warning": taper_warning}
app/ai_agents/fic_optimizer.py
class FICOptimizer:
    """
    Optimizes dividend strategy for FIC
    """
    def __init__(self, fic, dividend_options):
        self.fic = fic
        self.dividend_options = dividend_options

    def optimize(self):
        # Simple ranking by net efficiency
        ranked = sorted(self.dividend_options, key=lambda x: x['net_return'], reverse=True)
        return ranked[0:3]  # top 3 recommended strategies

import matplotlib.pyplot as plt

def capital_allocation_chart(data, labels, filename="chart.png"):
    fig, ax = plt.subplots()
    ax.pie(data, labels=labels, autopct='%1.1f%%')
    ax.set_title("Capital Allocation")
    plt.savefig(filename)
    return filename
